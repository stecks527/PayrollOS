Logger.log("PAYROLLWRITEROS WRITE"); Logger.log(JSON.stringify(result, null, 2)); return result; } /* ========================================================= * RENDER ONE PERSON / DAY * ========================================================= */ function payrollWriterOSRenderDay_( day ) { day = day || {}; /* * AEA DAY OFF always occupies the top row. */ const dayOff = calls.find( function (call) { return ( call.type === "AEA_DAY_OFF" ); } ); if (dayOff) { return { top: "AEA DAY OFF", bottom: "" }; } const rehearsal = calls.find( function (call) { return ( call.type === "REHEARSAL" || call.type === "REHEARSAL_NOT_CALLED" ); } ); let top = ""; let bottom = ""; if (matItems.length) { top = payrollWriterOSJoinPerformanceItems_( matItems, isTwoShowDay ); } if (eveItems.length) { bottom = payrollWriterOSJoinPerformanceItems_( eveItems, isTwoShowDay ); } } return { top: top, bottom: bottom }; } /* ========================================================= * PERFORMANCE ITEMS * ========================================================= */ function payrollWriterOSBuildPerformanceItems_( calls, outs ) { const items = []; } ); outs.forEach( function (out) { items.push({ type: "OUT", part: out.part || "Unknown", text: out.text || ( "OUT (" + ( out.reason || "Out" ) + ")" ), role: out.role || "", out: out.person || "" }); } ); return items.sort( function (a, b) { return ( payrollWriterOSPartRank_( a.part ) - payrollWriterOSPartRank_( b.part ) ); } ); } /* ========================================================= * JOIN PERFORMANCE ITEMS * ========================================================= */ function payrollWriterOSJoinPerformanceItems_( items, forcePartLabels ) { if (!items.length) { return ""; } /* * OUT and COVERAGE entries only need Mat/Eve when both * performances share one cell. */ if ( showPart && ( item.part === "Mat" || item.part === "Eve" ) ) { text += " - " + item.part; } return text; } function payrollWriterOSPartRank_( part ) { if (part === "Mat") { return 1; } if (part === "Eve") { return 2; } return 3; } /* ========================================================= * CELL WRITE * ========================================================= */ function payrollWriterOSSetCell_( range, text ) { text = String( text || "" );
